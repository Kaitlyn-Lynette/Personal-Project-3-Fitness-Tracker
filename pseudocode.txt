app.post("/update/:id", (req,res)=> {
  db.notes.update(
    {
      _id: mongojs.ObjectId(req.params.id)
    },
    {
      $set: {
        title: req.body.title,
        note: req.body.note,
        modified: Date.now()
      }
    },
    (error,data)=> {
      if(error) {
        res.send(error);
      } else{
        res.send(data);
      }
    }
  );
});
